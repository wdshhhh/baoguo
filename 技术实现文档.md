class Api::V1::AuthController < Api::V1::BaseController
  def login
    user = User.find_by(phone: params[:phone])
    if user&.authenticate(params[:password])
      token = JWT.encode({ user_id: user.id }, Rails.application.credentials.secret_key_base)
      render_json({ token: token, user: user.as_json(except: :password_digest) })

  end
end

### 3. 异常处理系统

#### 异常分类机制
```ruby
# app/models/package_exception.rb
class PackageException < ApplicationRecord
  enum exception_type: {
    damaged: 'damaged',              # 包裹损坏
    missing: 'missing',              # 包裹丢失
    wrong_address: 'wrong_address',  # 地址错误
    recipient_refused: 'recipient_refused' # 收件人拒收
  }
  
  enum status: {
    pending: 'pending',      # 待处理
    processing: 'processing',  # 处理中
    resolved: 'resolved'      # 已解决
  }
  
  # 关联包裹
  belongs_to :package
  belongs_to :created_by, class_name: 'User'
  belongs_to :resolved_by, class_name: 'User', optional: true
end
```

#### 异常解决流程
```ruby
# app/controllers/api/v1/exception_management_controller.rb
class Api::V1::ExceptionManagementController < Api::V1::BaseController
  def resolve
    exception = PackageException.find(params[:id])
    
    if exception.update(
      status: :resolved,
      resolved_by: current_user,
      resolved_at: Time.current,
      solution: params[:solution]
    )
      # 恢复包裹状态
      exception.package.update(status: :stored) if exception.package.exception?
      
      OperationLog.log("exception_resolved", user: current_user, resource: exception)
      render_json(format_exception(exception))
    else
      render_error("解决异常失败", errors: exception.errors.full_messages)
    end
  end
end
```

### 4. AI智能助手模块

#### DeepSeek API集成
```ruby
# app/services/deepseek_api_service.rb
class DeepseekApiService
  BASE_URL = "https://api.deepseek.com"
  
  def initialize(api_key = nil)
    @api_key = api_key || ENV['DEEPSEEK_API_KEY']
  end
  
  def chat(message, context = [])
    messages = context + [{ role: "user", content: message }]
    
    response = HTTParty.post(
      "#{BASE_URL}/chat/completions",
      headers: {
        "Authorization" => "Bearer #{@api_key}",
        "Content-Type" => "application/json"
      },
      body: {
        model: "deepseek-chat",
        messages: messages,
        max_tokens: 1000,
        temperature: 0.7
      }.to_json
    )
    
    if response.success?
      response.parsed_response["choices"][0]["message"]["content"]
    else
      "抱歉，AI助手暂时无法响应，请稍后再试"
    end
  end
end
```

# app/services/analytics_report_service.rb
class AnalyticsReportService
  def generate_smart_report(report_type, date_range = {})
    start_date = date_range[:start_date] || 30.days.ago
    end_date = date_range[:end_date] || Time.current
    
    case report_type
    when :daily_summary
      generate_daily_summary(start_date, end_date)
    when :weekly_trends
      generate_weekly_trends(start_date, end_date)
    when :monthly_analysis
      generate_monthly_analysis(start_date, end_date)
    when :exception_analysis
      generate_exception_analysis(start_date, end_date)
    when :performance_metrics
      generate_performance_metrics(start_date, end_date)
    else
      generate_comprehensive_report(start_date, end_date)
    end
  end
  
  # 周度趋势分析（使用groupdate gem）
  def generate_weekly_trends(start_date, end_date)
    weekly_data = Package.where(created_at: start_date..end_date)
                        .group_by_week(:created_at)
                        .count
    
    weekly_exceptions = PackageException.where(created_at: start_date..end_date)
                                       .group_by_week(:created_at)
                                       .count
    
    {
      success: true,
      data: {
        report_type: "weekly_trends",
        package_trends: weekly_data,
        exception_trends: weekly_exceptions,
        trend_analysis: analyze_weekly_trend(weekly_data)
      }
    }
  end
  
  # 异常分析报表
  def generate_exception_analysis(start_date, end_date)
    exceptions = PackageException.where(created_at: start_date..end_date)
    
    {
      success: true,
      data: {
        report_type: "exception_analysis",
        exception_stats: {
          total: exceptions.count,
          by_type: exceptions.group(:exception_type).count,
          by_status: exceptions.group(:status).count,
          resolution_time: calculate_avg_resolution_time(exceptions)
        },
        root_cause_analysis: analyze_exception_root_causes(exceptions),
        prevention_recommendations: generate_exception_prevention_recommendations(exceptions),
        high_risk_patterns: identify_high_risk_patterns(exceptions)

    }
  end
end

CREATE TABLE users (

#### packages表（包裹表）
```sql
CREATE TABLE packages (
  id INTEGER PRIMARY KEY AUTOINCREMENT,
  tracking_number VARCHAR(50) NOT NULL UNIQUE,
  pickup_code VARCHAR(20) NOT NULL UNIQUE,
  user_id INTEGER,
  recipient_name VARCHAR(100) NOT NULL,
  recipient_phone VARCHAR(20) NOT NULL,
  recipient_address TEXT NOT NULL,
  storage_location VARCHAR(50),
  status VARCHAR(20) DEFAULT 'pending',
  package_type VARCHAR(20) DEFAULT 'normal',
  weight DECIMAL(10,2),
  remark TEXT,
  stored_at DATETIME,
  picked_up_at DATETIME,
  stored_by_id INTEGER,
  picked_up_by_id INTEGER,
  created_at DATETIME,
  updated_at DATETIME

```

#### package_exceptions表（异常表）
```sql
CREATE TABLE package_exceptions (
  id INTEGER PRIMARY KEY AUTOINCREMENT,
  package_id INTEGER NOT NULL,
  exception_type VARCHAR(50) NOT NULL,
  description TEXT,
  status VARCHAR(20) DEFAULT 'pending',
  created_by_id INTEGER NOT NULL,
  resolved_by_id INTEGER,
  resolved_at DATETIME,
  solution TEXT,
  created_at DATETIME,
  updated_at DATETIME,
  FOREIGN KEY (package_id) REFERENCES packages(id),
  FOREIGN KEY (created_by_id) REFERENCES users(id),
  FOREIGN KEY (resolved_by_id) REFERENCES users(id)
);
```

#### Pinia状态管理
```javascript
// stores/auth.js
import { defineStore } from 'pinia'
import { ref } from 'vue'

export const useAuthStore = defineStore('auth', () => {
  const token = ref(localStorage.getItem('auth_token') || '')
  const user = ref(JSON.parse(localStorage.getItem('user_info') || '{}'))
  
  const login = async (phone, password) => {
    try {
      const response = await axios.post('/api/v1/auth/login', {
        phone,
        password
      })
      
      if (response.data.success) {
        token.value = response.data.data.token
        user.value = response.data.data.user
        
        localStorage.setItem('auth_token', token.value)
        localStorage.setItem('user_info', JSON.stringify(user.value))
        
        axios.defaults.headers.common['Authorization'] = `Bearer ${token.value}`
        
        return { success: true }
      }
    } catch (error) {
      return { success: false, message: error.response?.data?.error || '登录失败' }
    }
  }
  
  const logout = () => {
    token.value = ''
    user.value = {}
    localStorage.removeItem('auth_token')
    localStorage.removeItem('user_info')
    delete axios.defaults.headers.common['Authorization']
  }
  
  return { token, user, login, logout }
})
```
